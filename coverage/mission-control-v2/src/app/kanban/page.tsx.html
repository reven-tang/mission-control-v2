import { getTaskService } from '@/lib/services/task';
import type { TaskStatus, TaskPriority } from '@/lib/types';
 
const COLUMNS: { status: TaskStatus; label: string; badge: string; accent: string }[] = [
  { status: 'backlog',   label: '📦 Backlog',    badge: 'badge-gray',   accent: 'border-gray-400' },
  { status: 'todo',      label: '📝 To Do',      badge: 'badge-blue',   accent: 'border-blue-500' },
  { status: 'in_progress', label: '🚀 In Progress', badge: 'badge-amber',  accent: 'border-amber-500' },
  { status: 'review',    label: '👀 Review',     badge: 'badge-purple', accent: 'border-purple-500' },
  { status: 'done',      label: '✅ Done',       badge: 'badge-green',  accent: 'border-emerald-500' },
];
 
const PRIORITY_LABEL: Record<number, string> = { 0: '-', 1: 'P1', 2: 'P2', 3: 'P3', 4: 'P4' };
 
export default async function KanbanPage() {
  const kanban = await getTaskService().getKanban();
 
  return (
    <div className="min-h-screen bg-slate-100 dark:bg-slate-900">
      {/* Header */}
      <div className="sticky top-0 z-10 bg-white dark:bg-slate-800 border-b border-slate-200 dark:border-slate-700 px-6 py-4">
        <div className="flex items-center justify-between">
          <h1 className="text-xl font-bold text-slate-800 dark:text-white">📋 任务看板</h1>
          <div className="flex items-center gap-2 text-sm">
            <span className="px-2.5 py-1 bg-blue-50 dark:bg-blue-900/30 text-blue-600 dark:text-blue-400 rounded-md font-medium">
              总 {Object.values(kanban).flat().length} 个任务
            </span>
          </div>
        </div>
      </div>
 
      {/* Kanban board */}
      <div className="p-6 overflow-x-auto">
        <div className="flex gap-5 min-w-[1200px]">
          {COLUMNS.map(col => {
            const tasks = kanban[col.status] || [];
            return (
              <div key={col.status} className="flex-1 flex flex-col">
                {/* Column header */}
                <div className="flex items-center justify-between px-1 mb-3">
                  <div className="flex items-center gap-2">
                    <span className="text-base">{col.label}</span>
                    <span className="text-xs font-bold text-white bg-slate-400 dark:bg-slate-600 rounded-full w-5 h-5 flex items-center justify-center">
                      {tasks.length}
                    </span>
                  </div>
                </div>
 
                {/* Task list */}
                <div className="bg-slate-50 dark:bg-slate-800/50 rounded-xl border border-slate-200 dark:border-slate-700 p-2.5 space-y-2.5 flex-1 max-h-[calc(100vh-180px)] overflow-y-auto">
                  {tasks.map(task => (
                    <div
                      key={task.id}
                      className={`bg-white dark:bg-slate-800 rounded-lg border-l-[3px] ${col.accent} p-4 shadow-sm hover:shadow-md transition-shadow cursor-default group`}
                    >
                      <div className="flex items-start justify-between gap-2">
                        <h3 className="font-medium text-slate-800 dark:text-slate-100 text-sm leading-snug">
                          {task.title}
                        </h3>
                        <span className={`shrink-0 text-[10px] font-bold px-1.5 py-0.5 rounded
                          ${task.priority >= 3 ? 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400' :
                            task.priority === 2 ? 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400' :
                            'bg-slate-100 text-slate-500 dark:bg-slate-700 dark:text-slate-400'}`}>
                          {PRIORITY_LABEL[task.priority]}
                        </span>
                      </div>
 
                      {task.description && (
                        <p className="text-xs text-slate-500 dark:text-slate-400 mt-1.5 line-clamp-2 leading-relaxed">
                          {task.description}
                        </p>
                      )}
 
                      <div className="flex items-center gap-2 mt-3 text-[11px] text-slate-400 dark:text-slate-500">
                        <span className="px-1.5 py-0.5 bg-slate-100 dark:bg-slate-700 rounded">
                          {task.source === 'manual' ? '✋' : task.source === 'autonomous' ? '🤖' : '💡'} {task.source}
                        </span>
                        {task.assigned_agent && (
                          <span className="px-1.5 py-0.5 bg-blue-50 dark:bg-blue-900/30 text-blue-600 dark:text-blue-400 rounded">
                            🤖 {task.assigned_agent}
                          </span>
                        )}
                        {task.tags?.length ? (
                          <span className="truncate">{task.tags.slice(0, 2).map(t => (
                            <span key={t} className="ml-1 px-1.5 py-0.5 bg-slate-100 dark:bg-slate-700 rounded text-slate-500">
                              #{t}
                            </span>
                          ))}</span>
                        ) : null}
                      </div>
                    </div>
                  ))}
 
                  {tasks.length === 0 && (
                    <div className="py-8 text-center text-slate-300 dark:text-slate-600 text-sm">
                      暂无任务
                    </div>
                  )}
                </div>
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
}
