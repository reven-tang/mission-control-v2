import { createTask, getTask, updateTask, listTasks, getKanban, getStore } from '@/lib/db';
import { join } from 'path';
import type { Task, CreateTaskInput, TaskStatus } from '@/lib/types';
 
let instance: TaskService | null = null;
 
export function getTaskService(): TaskService {
  if (!instance) instance = new TaskService();
  return instance;
}
 
export class TaskService {
  async create(input: CreateTaskInput): Promise<Task> {
    return createTask(input);
  }
 
  async update(id: string, changes: Partial<Task>): Promise<Task> {
    return updateTask(id, changes);
  }
 
  async move(id: string, newStatus: TaskStatus): Promise<Task> {
    return updateTask(id, { status: newStatus });
  }
 
  async delete(id: string): Promise<void> {
    const s = getStore();
    s.tasks = s.tasks.filter((t: Task) => t.id !== id);
    import('fs').then(fs => {
      fs.writeFileSync(join(process.cwd(), 'data', 'mc-store.json'), JSON.stringify(s, null, 2));
    });
  }
 
  async getById(id: string): Promise<Task | null> {
    return getTask(id);
  }
 
  async list(filters?: { status?: string; priority?: number }): Promise<Task[]> {
    return listTasks(filters);
  }
 
  async getKanban() {
    return getKanban();
  }
}
